934 //
935 // 准备绘制透明图形（先绘制图形到临时画布中，然后再输出到需要绘制的地方）
936 // nGraphW, nGraphH 所绘制透明图形的宽高
937 //
938 // 注意：需要配合 DRAW_TNS_RENDER_TO 宏使用
939 //
940 // 使用方法：
941 // 在 DRAW_TNS_INIT_GRAPHICS 宏和 DRAW_TNS_RENDER_TO 宏之间，插入一个代码块。
942 // 在这个代码块中使用 Canvas 变量 graphics 进行绘制。
943 // 绘制时调用 Canvas 的普通绘图函数即可，无需 GDI+ 系列封装函数（带 "GP_" 前缀的函数）
944 // 或者直接使用原生 EasyX 函数进行绘制也可以。
945 //
946 // 使用示例：
947 /*
948  // 准备绘制透明图形（设置图形的宽高）
949  DRAW_TNS_INIT_GRAPHICS(201, 201);
950  {
951  // 在代码块中使用 Canvas 的普通绘图函数进行绘制即可
952  graphics.SetLineThickness(5);
953  graphics.FillRoundRect(0, 0, 200, 200, 20, 20, true, GREEN, PURPLE);
954 
955  // 像这样使用 EasyX 原生函数绘制也可以
956  line(20, 20, 50, 50);
957  }
958  // 最后选择将这个透明图形绘制到哪里，并设置绘制透明度
959  DRAW_TNS_RENDER_TO(120, 120, yourImagePointer, 100);
960 */
961 //
962 #define DRAW_TNS_INIT_GRAPHICS(nGraphW, nGraphH) \
963  {\
964  hiex::Canvas graphics(nGraphW, nGraphH);\
965  graphics.BeginBatchDrawing();(0)

967 //
968 // 完成绘制透明图形，并输出绘制的图形

971 // pDstImg 透明图形输出的目标画布（IMAGE*）

976 #define DRAW_TNS_RENDER_TO(nRenderX, nRenderY, pDstImg, alpha) \
977  graphics.EndBatchDrawing();\
