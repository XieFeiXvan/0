67  {

69  {
70  if (LOWORD(wParam) == GetID())
71  {

80  WCHAR* buf = new WCHAR[len + 1];
81  ZeroMemory(buf, (len + 1) * sizeof WCHAR);
82  ComboBox_GetLBText(GetHandle(), m_nSel, buf);
83  m_pFuncSel(m_nSel, buf);
84  delete[] buf;
85  }
86  break;
87 
88  case CBN_EDITUPDATE:

93  }
94  }
95  }

97  bRet = false;
